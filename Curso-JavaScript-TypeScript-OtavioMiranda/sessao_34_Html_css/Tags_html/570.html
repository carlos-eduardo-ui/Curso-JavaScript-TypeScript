        function gerarNumeroAleatorioJavaScript(minimo, maximo) {
  /**
   * Gera um número inteiro aleatório entre minimo (inclusive) e maximo (inclusive).
   */
  // Math.random() gera um número entre 0 e 1.
  // Multiplicamos pelo intervalo e somamos o mínimo para ajustar a faixa.
  // Usamos Math.floor() para arredondar para baixo e garantir um inteiro.
  return Math.floor(Math.random() * (maximo - minimo + 1)) + minimo;
}
